	-webkit-touch-callout: none;
	pointer-events: none; // 像微信浏览器还是无法禁止，加上这行样式即可

	-webkit-user-select: none;
}

	-webkit-touch-callout: none;
}

.border-right-1px,

		content: "";
		display: block;
		position: absolute;

	}
